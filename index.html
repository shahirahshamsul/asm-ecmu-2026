Live Sync Setup — takes about 5 minutes
1

Open Google Sheets

Go to sheets.google.com and create a new blank spreadsheet. Name it "Conference Attendance".

2

Open Apps Script

In your spreadsheet, click the menu: Extensions → Apps Script. Delete everything in the editor, then paste this code:

const SHEET_NAME = "Attendance"; function doGet(e) { const action = e.parameter.action; const sheet = getSheet(); if (action === "getAll") { const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const data = rows.slice(1).map(r => { const obj = {}; headers.forEach((h, i) => obj[h] = r[i]); return obj; }); return json(data); } return json({ error: "Unknown action" }); } function doPost(e) { const body = JSON.parse(e.postData.contents); const sheet = getSheet(); const action = body.action; if (action === "addAttendee") { const a = body.attendee; sheet.appendRow([ a.id, a.name, a.email, a.org||"", a.session, "false", "", "false", "", "false", "", new Date().toISOString(), "false", "" ]); return json({ success: true }); } if (action === "markEmailSent") { const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] === body.id) { sheet.getRange(i+1, 13).setValue("true"); sheet.getRange(i+1, 14).setValue(new Date().toISOString()); return json({ success: true }); } } return json({ error: "Not found" }); } if (action === "checkIn") { const day = body.day; const rows = sheet.getDataRange().getValues(); const dayColMap = { pre: [5,6], main1: [7,8], main2: [9,10], speaker: [7,8] }; const cols = dayColMap[day]; if (!cols) return json({ error: "Invalid day" }); for (let i = 1; i < rows.length; i++) { if (rows[i][0] === body.id) { const session = rows[i][4]; const allowed = isAllowed(session, day); if (!allowed) return json({ notAllowed: true, name: rows[i][1], session: session }); if (rows[i][cols[0]] === "true") return json({ duplicate: true, name: rows[i][1], day: day }); sheet.getRange(i+1, cols[0]+1).setValue("true"); sheet.getRange(i+1, cols[1]+1).setValue(new Date().toISOString()); return json({ success: true, name: rows[i][1], session: session }); } } return json({ error: "Not found" }); } return json({ error: "Unknown action" }); } function isAllowed(session, day) { if (day === "pre") return session === "Pre Congress" || session === "Both Congress"; if (day === "main1") return session === "Main Congress" || session === "Both Congress"; if (day === "main2") return session === "Main Congress" || session === "Both Congress"; if (day === "speaker") return session === "Speaker/VIP" || session === "Speaker / VIP" || session === "Facilitator"; // Speaker/VIP and Facilitator can scan on any day, once only if (session === "Speaker/VIP" || session === "Speaker / VIP" || session === "Facilitator") return true; return false; } function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow([ "id","name","email","org","session", "checkedIn_pre","checkInTime_pre", "checkedIn_main1","checkInTime_main1", "checkedIn_main2","checkInTime_main2", "createdAt","emailSent","emailSentTime" ]); sheet.setFrozenRows(1); } return sheet; } function json(data) { return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
